{

    int a,b;

    std::cin >> a;
    b = a/2;

    while (a%b != 0)
    {
        b--;
    }

    {
        std::cout << "Число " << a << " является простым" << std::endl;

    {

int exponA(int a, int b)
{
    if(b == 0)
    {
        return 1;
    }
    else if(b == 1)
    {
        return a;
    }
    else
    {
        b--;
        a *= exponA(a, b);
    }
}

// 02 B /////////////////////////////////////////

int exponB(int a, int b)
{
    if(b == 0)
    {
        return 1;
    }
    else if(b == 1)
    {
        return a;
    }
    else if(b%2 == 0)

        a *= a;
        b /= 2;
        a = exponB(a, b);

    else
    {

int ways(int** arr, int x, int y)
